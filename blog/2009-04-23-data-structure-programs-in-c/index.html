    if (front == -1)

}

    cout<<"\n Deleted " <<que[front];
    if (front == rear)
        front = rear = -1;
    else
        front++;
}

    if (front == -1) {
        cout<<"\n\t\t\t QUEUE EMPTY ";
        return;
    }
    cout<<"\n -----------------------------\n";
